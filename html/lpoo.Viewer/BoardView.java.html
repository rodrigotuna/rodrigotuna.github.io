package lpoo.Viewer;

import lpoo.GUI.GUI;
import lpoo.Model.Board;
import lpoo.Model.Position;

public class BoardView implements View<Board>{

    @Override
    public void draw(Board board, Position position, GUI gui) {
        TileViewFactory tileViewFactory = new TileViewFactory();
        TileView tileView = tileViewFactory.createView();

        Position tilePosition, tileScreenPosition;

        for(int line = 0; line < board.getHeight(); line++){
            for(int column = 0; column < board.getWidth(); column++){

                tilePosition = new Position(line,column);
                tileScreenPosition = new Position(column * 3 + position.getX() + 1, line * 2 + position.getY());

                tileView.draw(board.getTileAtPosition(tilePosition), tileScreenPosition, gui);
            }
        }

    }

}
