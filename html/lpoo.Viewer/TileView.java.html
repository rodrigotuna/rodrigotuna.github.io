package lpoo.Viewer;

import lpoo.GUI.GUI;
import lpoo.Model.Position;
import lpoo.Model.Tile;

public class TileView implements View<Tile>{
    @Override
    public void draw(Tile tile, Position position, GUI gui) {
        switch(tile.getStateName()){
            case OPEN:
                String adjacentMinesString = String.valueOf(tile.getAdjacentMines());
                gui.drawString(adjacentMinesString, position.getX(), position.getY(), "#FFCF99");
                break;
            case FLAGGED:
                gui.drawString("!", position.getX(), position.getY(), "#F15156");
                break;
            case CLOSED:
                gui.drawString("-", position.getX(), position.getY(), "#721121");
                break;
        }
    }
}
