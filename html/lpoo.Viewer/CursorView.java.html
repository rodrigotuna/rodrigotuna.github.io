package lpoo.Viewer;

import lpoo.GUI.GUI;
import lpoo.Model.Cursor;
import lpoo.Model.Position;

public class CursorView implements View<Cursor>{

    @Override
    public void draw(Cursor cursor, Position position, GUI gui) {
        int cursorScreenX = position.getX() + cursor.getPosition().getX();
        int cursorScreenY = position.getY() + cursor.getPosition().getY() * 2;

        gui.drawString("<",cursorScreenY, cursorScreenX,"#F57A00");
        gui.drawString(">",cursorScreenY + 2, cursorScreenX,"#F57A00");
    }
}
