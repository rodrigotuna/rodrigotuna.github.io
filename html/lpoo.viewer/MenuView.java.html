package lpoo.viewer;

import lpoo.gui.GUI;
import lpoo.model.Position;
import lpoo.model.menu.Menu;

public class MenuView extends View<Menu> {
    private static Menu menu;

    public MenuView(Menu menu) {
        super(menu);
    }

    public int drawTitle(GUI gui){
        String title = "minesweeper.";
        int center = gui.getWidthPos(title.length());
        for(int i = 0; i < title.length(); i++){
            String str = Character.toString(title.charAt(i));
            gui.drawString(str,center+i,4,getColor(i));
        }
        return center;
    }

    public void drawButtons(GUI gui, int center){
        gui.drawString("start game",center, gui.getHeightPos(-3), "#FFFFFF");
        gui.drawString("settings", center,gui.getHeightPos(-1), "#FFFFFF");
        gui.drawString("highscores", center,gui.getHeightPos(1), "#FFFFFF");
        gui.drawString("exit", center,gui.getHeightPos(3), "#FFFFFF");
    }

    @Override
    protected void drawElement(Position position, GUI gui) {
        gui.setTerminalBackgroundColor("#0F3325");

        FrameView frame = new FrameView(menu, gui.getDefaultMenuCols(), gui.getDefaultMenuRows());

        int center = drawTitle(gui);
        drawButtons(gui, center);

        frame.drawFrame(gui, new Position(0,0));
    }
}
