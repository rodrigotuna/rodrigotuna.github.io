package lpoo.viewer;

import lpoo.gui.GUI;
import lpoo.model.game.Cursor;
import lpoo.model.Position;

public class CursorView extends View<Cursor> {

    public CursorView(Cursor cursor) {
        super(cursor);
    }

    @Override
    protected void drawElement(Position position, GUI gui) {
        int cursorScreenX = position.getY() + getModel().getPosition().getY()* 3;
        int cursorScreenY = position.getX() + getModel().getPosition().getX()* 2;

        gui.drawString("<",cursorScreenX, cursorScreenY,"#FFFFFF");
        gui.drawString(">",cursorScreenX + 2, cursorScreenY,"#FFFFFF");
    }
}
