package lpoo.gui;

public class Action {
    GUI.ACTION action;
    Character key;

    public Action(GUI.ACTION action, Character key){
        this.action = action;
        this.key = key;
    }

    public GUI.ACTION getAction() {
        return action;
    }

    public Character getKey() {
        return key;
    }
}
