package lpoo.Model;

public class Game {
    private Board board;
    private Cursor cursor;

    public Game(Board board){
        this.board = board;
        this.cursor = new Cursor(0,0, board.getHeight(), board.getWidth());
    }

    public Board getBoard() {
        return board;
    }

    public Cursor getCursor() {
        return cursor;
    }
}
