package lpoo.model.menu;

import lpoo.model.Configuration;
import lpoo.model.Element;

import java.util.Arrays;
import java.util.List;

public class SettingsMenu implements Element {
    private final List<String> entries;
    private int currentEntry = 0;
    private Configuration configuration;
    private List<Integer> configurationValues;

    public SettingsMenu(Configuration configuration) {
        this.entries = Arrays.asList("board height", "board width", "number of mines", "back");
        this.configuration = configuration;
        this.configurationValues = Arrays.asList(
                configuration.getHeight(),configuration.getWidth(),configuration.getNumMines());
    }

    private void update(){
        this.configurationValues = Arrays.asList(
                configuration.getHeight(),configuration.getWidth(),configuration.getNumMines());
    }

    public void nextEntry() {
        currentEntry++;
        if (currentEntry > this.entries.size() - 1)
            currentEntry = 0;
    }

    public void previousEntry() {
        currentEntry--;
        if (currentEntry < 0)
            currentEntry = this.entries.size() - 1;
    }

    public String getEntry(int i) {
        return entries.get(i);
    }

    public boolean isSelected(int i) {
        return currentEntry == i;
    }

    public boolean isSelectedHeight() {
        return isSelected(1);
    }

    public boolean isSelectedWidth() {
        return isSelected(0);
    }

    public boolean isSelectedMines() {return isSelected(2);}

    public boolean isSelectedBack() {return isSelected(3);}

    public int getNumberEntries() {
        return this.entries.size();
    }

    public boolean getSelected(int index){
        return isSelected(index);
    }

    public Configuration getConfiguration(){
        return configuration;
    }

    public String getConfigurationValue(int index) {
        String valueStr =  String.valueOf(configurationValues.get(index));
        if(valueStr.length() == 1) valueStr = " " + valueStr;
        return valueStr;
    }

    public void increase(){
        switch(currentEntry){
            case 0:
                configuration.setHeight(configuration.getHeight() + 1);
                break;
            case 1:
                configuration.setWidth(configuration.getWidth() + 1);
                break;
            case 2:
                configuration.setNumMines(configuration.getNumMines() + 1);
                break;
        }

        update();
    }

    public void decrease(){
        switch(currentEntry) {
            case 0:
                configuration.setHeight(configuration.getHeight() - 1);
                break;
            case 1:
                configuration.setWidth(configuration.getWidth() - 1);
                break;
            case 2:
                configuration.setNumMines(configuration.getNumMines() - 1);
                break;
        }

        update();
    }
}
