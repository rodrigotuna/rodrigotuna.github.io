package lpoo.Controller.Commands;

import lpoo.Model.Game;

public class FlagCommand extends TileCommand {
    public FlagCommand(Game game) {
        super(game);
    }

    @Override
    public void execute() {
        selectedTile().setState(selectedTile().getState().flag());
    }
}
