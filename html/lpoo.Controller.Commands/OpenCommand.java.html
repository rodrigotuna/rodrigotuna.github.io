package lpoo.Controller.Commands;

import lpoo.Model.Board;
import lpoo.Model.Game;
import lpoo.Model.Position;
import lpoo.Model.Tile;
import lpoo.Model.TileState.TileState;

public class OpenCommand extends TileCommand {

    public OpenCommand(Game game) {
        super(game);
    }

    private Position getCursorPosition(){
        return game.getCursor().getPosition();
    }

    private Board getGameBoard(){
        return game.getBoard();
    }

    private Tile getTileAtPosition(int x, int y){
        return getGameBoard().getTileAtPosition(new Position(x,y));
    }

    private void cascadeOpen(int x, int y){
        if(getTileAtPosition(x,y).getAdjacentMines() == 0){
            for(int line = Math.max(0, x - 1); line < Math.min(getGameBoard().getHeight(), x + 2); line++){
                for(int col = Math.max(0, y - 1); col < Math.min(getGameBoard().getHeight(), y + 2); col++){
                    if(getTileAtPosition(line,col).getStateName() == TileState.STATE.CLOSED)
                    cascadeOpen(line,col);
                }
            }
        }
        getTileAtPosition(x,y).setState(getTileAtPosition(x,y).getState().open());
    }

    @Override
    public void execute() {
        selectedTile().setState(selectedTile().getState().open());
    }
}
