package lpoo.Controller;

import lpoo.Controller.Commands.*;
import lpoo.GUI.GUI;
import lpoo.Model.Game;
import lpoo.Model.Position;
import lpoo.Viewer.BoardView;
import lpoo.Viewer.CursorView;

import java.io.IOException;

public class GameController {
    private Game game;
    private GUI gui;

    public GameController(Game game, GUI gui){
        this.game = game;
        this.gui = gui;
    }

    public void start() throws IOException, InterruptedException {
        boolean running = true;
        while(running){
            switch(gui.getNextAction()){
                case UP:
                    new MoveUpCommand(game).execute();
                    break;
                case DOWN:
                    new MoveDownCommand(game).execute();
                    break;
                case LEFT:
                    new MoveLeftCommand(game).execute();
                    break;
                case RIGHT:
                    new MoveRightCommand(game).execute();
                    break;
                case FLAG:
                    new FlagCommand(game).execute();
                    break;
                case OPEN:
                    new OpenCommand(game).execute();
                    break;
                case QUIT:
                    running = false;
                    break;
            }
            gui.clear();
            new CursorView().draw(game.getCursor(), game.getCursor().getPosition(), gui);
            new BoardView().draw(game.getBoard(), new Position(0,0), gui);
            gui.refresh();
            Thread.sleep(100);
        }
        gui.close();
    }

}
